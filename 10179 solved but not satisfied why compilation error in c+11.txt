#define maxx 31700
bool flag[maxx+1]={false};


vi v;

void generate_prime()
{
    v.pb(2);

    int limit=sqrt(maxx)+2;
    for(int i=3;i<=maxx;i+=2)
    {

        if(flag[i]==false)
        {
            v.pb(i);
            if(i<=limit)
            {
                for(ull j=i*i;j<=maxx;j+=(i*2))
                {
                    flag[j]=true;
                }
            }
        }

    }
}
int siz;
void magic_box(ull n)
{
    int index=0;
    int ans=n;
    set<int>s;
    while(n>1)
    {
        if(index==siz)break;

        if(n%v[index]==0)
        {
            n=n/v[index];
            s.insert(v[index]);
        }
        else
        {
            index++;
        }
    }
    if(n>1)
    {
        s.insert(n);
    }

    for (std::set<int>::iterator it=s.begin(); it!=s.end(); ++it)
    {
        ans=ans/(*it);
        ans=ans*(*it-1);
    }
    cout<<ans<<endl;
}
int main()
{
    //read("incb.txt");
    //write("outcb.txt");
    generate_prime();
    siz=v.size();
    ull n;
    while(true)
    {
        cin>>n;
        if(n==0)break;
        magic_box(n);
    }
}
